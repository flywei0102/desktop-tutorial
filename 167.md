class Solution {
    public int[] twoSum(int[] numbers, int target) {
        int left = 0;
        int right = numbers.length - 1; 
        
        int[] output = {-1, -1};
        
        
        while(left < right){
            int sum = numbers[left] + numbers[right];
            if(sum == target){
                output[0] = left + 1;
                output[1] = right + 1;
                return output;
            }else if(sum < target){
                left ++;
            }else{
                right --;
            }
        }
        
        return output;
    }
}
