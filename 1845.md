class SeatManager {
    PriorityQueue<Integer> heap;

    public SeatManager(int n) {
        heap = new PriorityQueue<>(n);
        for(int i = 1; i <= n; i++){
            heap.offer(i);
        }
    }
    
    public int reserve() {
        return heap.poll();
    }
    
    public void unreserve(int seatNumber) {
        heap.offer(seatNumber);
    }
}
