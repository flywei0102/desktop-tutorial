class Solution {
    public int[] sortArrayByParityII(int[] nums) {
        int j = 1;  // nums[j] is odd
        for(int i = 0; i < nums.length; i += 2){
            if(nums[i] % 2 != 0){
                while(nums[j] % 2 == 1){
                    j += 2;
                }
                swap(nums, i, j);
            }
        }
        return nums;
    }



    private void swap(int[] arr, int i, int j){
        int temp = arr[i];
        arr[i] = arr[j];
        arr[j] = temp;
    }

    
}
